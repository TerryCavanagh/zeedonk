[1,2,3].indexOf(x:T, fromIndex:Int = 0):Int
Returns the index of the first occurrence of x in the array.

If it's not found, returns -1. If fromIndex is supplied, it will start its search from this point and move right. If fromIndex is negative, it's treated as "this.length-fromIndex".

a = ["a","b","c","a","b","c"];
trace(a.indexOf("c"));

[1,2,3].lastIndexOf(x:T, fromIndex:Int = this.length-1):Int
Returns the index of the last occurence of x in the Array, searching right to left.

If x is not found, the function returns -1.

If fromIndex is given, it will start from this point and go left, otherwise it starts from the last element. If it's negative, it's as if you passed "this.length-fromIndex".

a = ["a","b","c","a","b","c"];
trace(a.lastIndexOf("b"));

[1,2,3].slice(pos:Int, end:Int = this.length):Array
Creates a new array with elements from the old one starting from pos, and up to (but not including) end.

The elements themselves are not copied.

If you don't give an end it'll go to the end of the array. If the either of the arguments are negative, counts back from the end of the array (this.length + pos, this.length + end).

var a = ["cat","ball","balloon","crying","parents","rain"];
var b = a.slice(2,4);
trace(b);

//count number of 'L's in a string
function lCount(s){
  var count=0;
  for(i in 0...s.length){
    if (s[i]=="l"||s[i]=='L'){
      count++;
    }
  }    
  return count;
}

//sort by number of 'L's
function compare(a,b){
  return lCount(a)-lCount(b);
}

[1,2,3].splice(pos:Int, len:Int):Array
Removes len elements from the array, starting at pos, returning the elements removed in an array.

This modifies the array.

If len is negative, or too large large, an empty array is returned

If pos is negative, it's treated as this.length-pos.

If len+pos>this.length, then it will remove until the end of the array.

Super useful for removing elements from an array, if you know the index.

var a = ["cat","ball","balloon","crying","parents","rain"];
var b = a.splice(2,2);
trace(b);
